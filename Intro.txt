pcall(function() if game.CoreGui:FindFirstChild("BrutonIntro") then game.CoreGui.BrutonIntro:Destroy() end end)

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "BrutonIntro"
ScreenGui.Parent = game.CoreGui
ScreenGui.ResetOnSpawn = false

local TweenService = game:GetService("TweenService")

local bg = Instance.new("Frame")
bg.Size = UDim2.new(1,0,1,0)
bg.Position = UDim2.new(0,0,0,0)
bg.BackgroundColor3 = Color3.fromRGB(255,120,0)
bg.BackgroundTransparency = 0.6
bg.BorderSizePixel = 0
bg.Parent = ScreenGui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(0.6,0,0.15,0)
title.Position = UDim2.new(0.2,0,0.2,0)
title.BackgroundTransparency = 1
title.Text = "BRUTON HUB"
title.TextColor3 = Color3.fromRGB(255,255,255)
title.Font = Enum.Font.GothamBold
title.TextScaled = true
title.Parent = bg

local pulse = true
task.spawn(function()
	while pulse do
		local tween1 = TweenService:Create(title,TweenInfo.new(0.8,Enum.EasingStyle.Sine,Enum.EasingDirection.InOut),{TextColor3 = Color3.fromRGB(255,200,50)})
		local tween2 = TweenService:Create(title,TweenInfo.new(0.8,Enum.EasingStyle.Sine,Enum.EasingDirection.InOut),{TextColor3 = Color3.fromRGB(255,255,255)})
		tween1:Play()
		tween1.Completed:Wait()
		tween2:Play()
		tween2.Completed:Wait()
	end
end)

local barFrame = Instance.new("Frame")
barFrame.Size = UDim2.new(0.6,0,0.05,0)
barFrame.Position = UDim2.new(0.2,0,0.55,0)
barFrame.BackgroundColor3 = Color3.fromRGB(50,50,50)
barFrame.BorderSizePixel = 0
barFrame.ClipsDescendants = true
barFrame.Parent = bg

local barCorner = Instance.new("UICorner")
barCorner.CornerRadius = UDim.new(0.5,0)
barCorner.Parent = barFrame

local bar = Instance.new("Frame")
bar.Size = UDim2.new(0,0,1,0)
bar.Position = UDim2.new(0,0,0,0)
bar.BackgroundColor3 = Color3.fromRGB(255,200,0)
bar.BorderSizePixel = 0
bar.Parent = barFrame
local barCorner2 = Instance.new("UICorner")
barCorner2.CornerRadius = UDim.new(0.5,0)
barCorner2.Parent = bar

local head = Instance.new("TextLabel")
head.Size = UDim2.new(0.05,0,1,0)
head.Position = UDim2.new(0,0,0,0)
head.BackgroundTransparency = 1
head.Text = "ðŸŽƒ"
head.TextScaled = true
head.Parent = bar

task.spawn(function()
	while barFrame.Parent do
		local tween1 = TweenService:Create(bar,TweenInfo.new(0.8,Enum.EasingStyle.Sine,Enum.EasingDirection.InOut),{BackgroundColor3 = Color3.fromRGB(255,150,0)})
		local tween2 = TweenService:Create(bar,TweenInfo.new(0.8,Enum.EasingStyle.Sine,Enum.EasingDirection.InOut),{BackgroundColor3 = Color3.fromRGB(255,200,0)})
		tween1:Play()
		tween1.Completed:Wait()
		tween2:Play()
		tween2.Completed:Wait()
	end
end)

local duration = 6
local steps = 100
local stepTime = duration/steps

for i = 1, steps do
	local newSize = UDim2.new(i/steps,0,1,0)
	bar:TweenSize(newSize,"Out","Linear",stepTime,true)
	head:TweenPosition(UDim2.new(i/steps,0,0,0),"Out","Linear",stepTime,true)
	title.Text = "BRUTON HUB - "..i.."%"
	task.wait(stepTime)
end

pulse = false
ScreenGui:Destroy()
